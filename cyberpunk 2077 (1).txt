    <script>
        // Typing effect for the terminal
        document.addEventListener('DOMContentLoaded', function() {
            const text = "Hello, I am Vinuta!";
            const typingElement = document.getElementById('typing-text');
            let i = 0;
            
            function typeWriter() {
                if (i < text.length) {
                    typingElement.innerHTML += text.charAt(i);
                    i++;
                    setTimeout(typeWriter, 100);
                }
            }
            
            typeWriter();
            
            // Form submission
            const contactForm = document.getElementById('contactForm');
            if (contactForm) {
                contactForm.addEventListener('submit', function(e) {
                    e.preventDefault();
                    alert('Message transmitted through the cyber network!');
                    contactForm.reset();
                });
            }
            
            // Add parallax effect to cityscape
            window.addEventListener('scroll', function() {
                const scrollY = window.scrollY;
                const cityscape = document.querySelector('.cityscape');
                if (cityscape) {
                    cityscape.style.transform = `translateY(${scrollY * -0.2}px)`;
                }
            });
        });
    </script>
